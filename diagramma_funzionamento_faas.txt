\begin{figure}[htbp]
    \centering
    \makebox[\textwidth][c]{%
    \begin{tikzpicture}[
        node distance=1.4cm and 0.35cm,
        box/.style={draw=black!70, thick, rectangle, rounded corners,
                    minimum width=4.5cm, minimum height=1.1cm,
                    align=center, fill=blue!5},
        note/.style={align=left, font=\small, text=black!80, text width=4.5cm},
        note_left/.style={align=right, font=\small, text=black!80, text width=4.5cm},
        arrow/.style={-{Stealth[length=3mm, width=2mm]}, thick},
        branch/.style={-{Stealth[length=2.5mm, width=1.5mm]}, thick, draw=black!60}
    ]

    % --- Nodi principali ---
    \node[box, fill=orange!5] (evento)
        {\textbf{Generazione Evento}\\\texttt{Richiesta Utente o Trigger}};
    \node[note, right=0.35cm of evento]
        {$\longleftarrow$ Un'azione dell'utente o un sistema esterno (es. chiamata web, modifica dati) avvia il flusso.};

    \node[box, fill=blue!5, below=1.4cm of evento] (monitor)
        {\textbf{Rilevamento Evento}\\\texttt{Router / Smistatore}};
    \node[note, right=0.35cm of monitor]
        {$\longleftarrow$ Il punto d'accesso cattura l'evento in tempo reale per capire quale funzione debba rispondere.};

    \node[box, fill=yellow!5, below=1.4cm of monitor] (api)
        {\textbf{Richiesta di Avvio}\\\texttt{Invocazione Piattaforma}};
    \node[note, right=0.35cm of api]
        {$\longleftarrow$ Il sistema contatta le interfacce di controllo della piattaforma per preparare ed eseguire il codice.};

    \node[box, fill=green!5, below=1.4cm of api] (avvio)
        {\textbf{Avvio della Funzione}\\\texttt{Attivazione Istantanea}};
    \node[note, right=0.35cm of avvio]
        {$\longleftarrow$ Il codice si attiva in pochissimi millisecondi, rendendo l'attesa impercettibile.};

    \node[box, fill=purple!5, below=1.4cm of avvio] (stateless)
        {\textbf{Risultato ed Elaborazione}\\\texttt{Calcolo Stateless}};
    \node[note, right=0.35cm of stateless]
        {$\longleftarrow$ La funzione elabora la risposta e si spegne subito dopo. Non conserva dati in memoria per le esecuzioni future.};

    % --- Frecce e Ramificazioni ---

    % evento → monitor
    \draw[arrow] (evento) -- (monitor) coordinate[midway] (e_m);
    \draw[branch] (e_m) -- +(-2.6cm, 0) node[left, note_left] {Rilevamento del segnale di trigger};

    % monitor → api
    \draw[arrow] (monitor) -- (api) coordinate[midway] (m_a);
    \draw[branch] (m_a) -- +(-2.6cm, 0) node[left, note_left] {Identificazione e routing della richiesta};

    % api → avvio
    \draw[arrow] (api) -- (avvio) coordinate[midway] (a_av);
    \draw[branch] (a_av) -- +(-2.6cm, 0) node[left, note_left] {Allocazione rapida del codice in esecuzione};

    % avvio → stateless
    \draw[arrow] (avvio) -- (stateless) coordinate[midway] (av_s);
    \draw[branch] (av_s) -- +(-2.6cm, 0) node[left, note_left] {Esecuzione della logica e ritorno del risultato};

    \end{tikzpicture}%
    }
    \caption{Architettura e flusso logico concettuale del funzionamento di un servizio FaaS (Function-as-a-Service)}
    \label{fig:faas-lifecycle-general}
\end{figure}
