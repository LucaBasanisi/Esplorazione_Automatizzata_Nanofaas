\begin{figure}[htbp]
    \centering
    \begin{tikzpicture}[
        scale=0.9, every node/.style={transform shape},
        node distance=1.4cm and 0.4cm,
        box/.style={draw=black!70, thick, rectangle, rounded corners,
                    minimum width=5.0cm, minimum height=1.1cm,
                    align=center, fill=blue!5},
        note/.style={align=left, font=\small, text=black!80, text width=5.0cm},
        note_left/.style={align=right, font=\small, text=black!80, text width=5.0cm},
        arrow/.style={-{Stealth[length=3mm, width=2mm]}, thick},
        branch/.style={-{Stealth[length=2.5mm, width=1.5mm]}, thick, draw=black!60}
    ]

    % --- Nodi principali ---
    \node[box, fill=orange!5] (just)
        {\textbf{Innesco Test}\\\texttt{just <nome-test>}};
    \node[note, right=0.4cm of just]
        {$\longleftarrow$ Il tester avvia un target specifico (es. \texttt{just test-ping-pong}).};

    \node[box, below=1.4cm of just] (lib)
        {\textbf{Importazione Iniziale}\\\texttt{source e2e-utils.sh}};
    \node[note, right=0.4cm of lib]
        {$\longleftarrow$ Caricamento immediato di variabili globali, colori e funzioni di controllo VM.};

    \node[box, fill=red!5, below=1.4cm of lib] (setup)
        {\textbf{Orchestratore Sandbox}\\\texttt{e2e-external-api-setup.sh}};
    \node[note, right=0.4cm of setup]
        {$\longleftarrow$ Gestisce lo stato della VM. Esegue il ripristino distruttivo dello snapshot pulito.};

    \node[box, fill=yellow!5, below=1.4cm of setup] (compile)
        {\textbf{Deploy Condizionale}\\\texttt{setup-nanofaas.sh}};
    \node[note, right=0.4cm of compile]
        {$\longleftarrow$ Viene invocato solo in assenza di snapshot per compilare ed esporre nanoFaaS da zero.};

    \node[box, below=1.4cm of compile] (k8s)
        {\textbf{Verifica \& Port-Forward}\\\texttt{kubectl wait \& systemd-run}};
    \node[note, right=0.4cm of k8s]
        {$\longleftarrow$ Attesa dello stato \texttt{Ready} di k3s e avvio del tunnel per le chiamate API.};

    \node[box, fill=green!5, below=1.4cm of k8s] (exec)
        {\textbf{Esecuzione Test}\\\texttt{e2e-<nome-test>.sh}};
    \node[note, right=0.4cm of exec]
        {$\longleftarrow$ Compilazione della funzione sotto test ed invio dei payload d'attacco.};

    \node[box, fill=purple!5, below=1.4cm of exec] (verify)
        {\textbf{Verifica Risultati}\\\texttt{grep, logs \& metrics}};
    \node[note, right=0.4cm of verify]
        {$\longleftarrow$ Ispezione automatica dei log del Control Plane e dei pod per validare le euristiche.};

    \node[box, fill=red!5, below=1.4cm of verify] (cleanup)
        {\textbf{Teardown \& Cleanup}\\\texttt{trap cleanup EXIT}};
    \node[note, right=0.4cm of cleanup]
        {$\longleftarrow$ Rimozione automatica delle funzioni deployate e pulizia dei file temporanei sulla VM.};

    % --- Frecce e Ramificazioni ---

    % just → lib
    \draw[arrow] (just) -- (lib) coordinate[midway] (j_l);
    \draw[branch] (j_l) -- +(-2.9cm, 0) node[left, note_left] {Avvio script di test ed export configurazione};

    % lib → setup
    \draw[arrow] (lib) -- (setup) coordinate[midway] (l_s);
    \draw[branch] (l_s) -- +(-2.9cm, 0) node[left, note_left] {Invocazione dello script di setup preliminare};

    % setup → compile
    \draw[arrow] (setup) -- (compile) coordinate[midway] (s_c);
    \draw[branch] (s_c) -- +(-2.9cm, 0) node[left, note_left] {\texttt{multipass restore} o build da zero};

    % compile → k8s
    \draw[arrow] (compile) -- (k8s) coordinate[midway] (c_k);
    \draw[branch] (c_k) -- +(-2.9cm, 0) node[left, note_left] {Controllo connettività ed avvio port-forward};

    % k8s → exec
    \draw[arrow] (k8s) -- (exec) coordinate[midway] (k_e);
    \draw[branch] (k_e) -- +(-2.9cm, 0) node[left, note_left] {Esecuzione della logica specifica dell'esperimento};

    % exec → verify
    \draw[arrow] (exec) -- (verify) coordinate[midway] (e_v);
    \draw[branch] (e_v) -- +(-2.9cm, 0) node[left, note_left] {Parsing dei log ed estrazione metriche};

    % verify → cleanup
    \draw[arrow] (verify) -- (cleanup) coordinate[midway] (v_cl);
    \draw[branch] (v_cl) -- +(-2.9cm, 0) node[left, note_left] {Ripristino delle condizioni iniziali del cluster};

    \end{tikzpicture}
    \caption{Ciclo di vita completo di una sessione di test esplorativo assistito da strumenti}
    \label{fig:scripts-lifecycle}
\end{figure}
